## Phase 1: Design Decisions

### Agent Architecture

Both agents use `@openai/agents` (Agent/Runner abstraction) for Groq levels and `@google/generative-ai` for the Gemini fallback. The `@openai/agents` SDK requires `openai` as a peer dep for the underlying HTTP client.

```typescript
import { Agent, run } from '@openai/agents'
import OpenAI from 'openai'
import { GoogleGenerativeAI } from '@google/generative-ai'

const groqClient = new OpenAI({
  baseURL: 'https://api.groq.com/openai/v1',
  apiKey: process.env.GROQ_API_KEY
})

export async function generateCareerGuidance(assessment: AssessmentInput): Promise<CareerGuidance> {
  // Level 1: @openai/agents → Groq llama-3.3-70b-versatile
  try {
    const agent = new Agent({
      name: 'CareerGuidanceAgent',
      instructions: buildPrompt(assessment),
      model: 'llama-3.3-70b-versatile',
      outputType: CareerGuidanceSchema   // Zod schema — structured output
    })
    const result = await run(agent, 'Generate career guidance', { /* groqClient */ })
    return CareerGuidanceSchema.parse(result.finalOutput)
  } catch { /* fall through */ }

  // Level 2: @openai/agents → Groq llama-3.1-8b-instant (same agent, lighter model)
  try { /* same Agent pattern, model: 'llama-3.1-8b-instant' */ } catch { /* fall through */ }

  // Level 3: @google/generative-ai → Gemini 1.5 Flash (direct, JSON mode)
  try { /* existing @google/generative-ai pattern with responseMimeType: 'application/json' */ } catch { /* fall through */ }

  throw new Error('All AI providers failed')
}
```

### Sanitization

```typescript
const INJECTION_PATTERNS = [
  /ignore\s+(all\s+|previous\s+)?instructions/gi,
  /disregard\s+(all\s+|the\s+)?instructions/gi,
  /you\s+are\s+now/gi,
  /act\s+as/gi,
  /system\s*:/gi,
  /forget\s+everything/gi,
  /new\s+instructions/gi,
  /```/g,
]

export function sanitizeInput(text: string, maxLength = 4000): string {
  let cleaned = text
  for (const pattern of INJECTION_PATTERNS) {
    cleaned = cleaned.replace(pattern, '')
  }
  return cleaned.trim().slice(0, maxLength)
}
```

```
